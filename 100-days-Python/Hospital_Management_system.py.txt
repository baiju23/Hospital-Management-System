        print("1. Add Patient")

        print("3. Staff Details")
        print("4. Exit")
        print("5. Search Patient")

            age = int(input("Patient Age : "))
            disease = input("Disease Name : ")

            patient = {

            search_name = input("Enter Patient Name : ")

            found = False

            for patient in patients:

                if patient["Name"] == search_name:

            if found == False:
                print("Patient Not Found")

        elif choice == 6:

            delete_name = input("Enter Patient Name to Delete : ")

            found = False

                if patient["Name"] == delete_name:

                    patients.remove(patient)

                    found = True

                    break

            if found == False:
                print("Patient Not Found")
